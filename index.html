from flask import Flask, request, jsonify, render_template_string import base64 import re import subprocess import sys app = Flask(__name__) # -------------------- دوال فك التشفير -------------------- def decode_ascii_escapes(text): """فك \072\065\073 إلى أحرف""" def replace(match): return chr(int(match.group(1))) return re.sub(r'\\(\d+)', replace, text) def xor_decrypt(data, key): """فك XOR""" result = "" for i in range(len(data)): data_byte = ord(data[i]) key_byte = ord(key[i % len(key)]) result += chr(data_byte ^ key_byte) return result def extract_xor_v7(text): """استخراج v7("data","key") من النص""" pattern = r'v7\(\s*["\']([^"\']+)["\']\s*,\s*["\']([^"\']+)["\']\s*\)' matches = re.findall(pattern, text) return matches def try_common_keys(data): """تجربة مفاتيح شائعة""" common_keys = [ r"\126\177\163\187\69\134\219\167", r"\38\84\215\41\118\220\70", r"\156\67\173\74\165", r"\158\48\118\66\114", r"\38\156\55\199", "KEY_DARK_MLNJAVB", "secret" ] for key in common_keys: decoded_key = decode_ascii_escapes(key) try: result = xor_decrypt(data, decoded_key) if result.isprintable() and len(result) > 10: return result except: continue return None def decode_base64(text): """فك Base64""" try: return base64.b64decode(text).decode('utf-8', errors='ignore') except: return None def decode_hex(text): """فك Hex""" try: return bytes.fromhex(text).decode('utf-8', errors='ignore') except: return None def full_decode(script): """فك التشفير الكامل بكل الطرق""" result = script changed = True iterations = 0 max_iter = 10 # فك ASCII escapes result = decode_ascii_escapes(result) while changed and iterations < max_iter: changed = False new_result = result # 1. فك v7 XOR xor_matches = extract_xor_v7(new_result) for data, key in xor_matches: data_decoded = decode_ascii_escapes(data) key_decoded = decode_ascii_escapes(key) decrypted = xor_decrypt(data_decoded, key_decoded) # جرب فك Base64 b64 = decode_base64(decrypted) if b64: decrypted = b64 pattern = f'v7("{data}","{key}")' if decrypted and decrypted != pattern: new_result = new_result.replace(pattern, decrypted) changed = True # 2. فك Base64 b64_pattern = r'[A-Za-z0-9+\/=]{30,}' b64_matches = re.findall(b64_pattern, new_result) for b64_str in b64_matches: decoded = decode_base64(b64_str) if decoded and len(decoded) > 10 and not any(c in decoded for c in ['\x00', '\xff']): new_result = new_result.replace(b64_str, decoded) changed = True # 3. فك Hex hex_pattern = r'[0-9A-Fa-f]{32,}' hex_matches = re.findall(hex_pattern, new_result) for hex_str in hex_matches: decoded = decode_hex(hex_str) if decoded and len(decoded) > 5: new_result = new_result.replace(hex_str, decoded) changed = True # 4. تجربة مفاتيح شائعة على أي نص طويل long_strings = re.findall(r'["\']([^"\']{30,})["\']', new_result) for long_str in long_strings: decoded = try_common_keys(long_str) if decoded: new_result = new_result.replace(f'"{long_str}"', f'"{decoded}"') changed = True result = new_result iterations += 1 return result # -------------------- واجهة HTML -------------------- HTML_TEMPLATE = ''' فك تشفير سكريبتات Roblox (كامل)

''' @app.route('/') def index(): return render_template_string(HTML_TEMPLATE) @app.route('/decode', methods=['POST']) def decode(): data = request.json script = data.get('script', '') if not script: return jsonify({'success': False, 'error': 'لا يوجد نص'}) try: result = full_decode(script) return jsonify({'success': True, 'result': result}) except Exception as e: return jsonify({'success': False, 'error': str(e)}) @app.route('/retry', methods=['POST']) def retry(): data = request.json script = data.get('script', '') if not script: return jsonify({'success': False, 'error': 'لا يوجد نص'}) try: # طرق إضافية: تكرار أكثر، محاولة فك كل سلسلة result = script for i in range(15): # تكرار أكثر new_result = full_decode(result) if new_result == result: break result = new_result return jsonify({'success': True, 'result': result}) except Exception as e: return jsonify({'success': False, 'error': str(e)}) if __name__ == '__main__': app.run(host='0.0.0.0', port=5000, debug=True)
